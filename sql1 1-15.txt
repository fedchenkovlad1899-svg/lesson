(id  INTEGER PRIMARY KEY AUTOINCREMENT,
age  INTEGER,
sex  TEXT CHECK(sex = 'male' or sex = 'female'),
country  TEXT,
name TEXT NOT NULL,
last_name  TEXT NOT NULL
);

9.	Вывести актрис из Франции младше 35 лет.

10.	Вывести 5 самых молодых актёров.
SELECT * FROM actors 
WHERE age NOT NULL AND sex = 'male'
ORDER BY age ASC
LIMIT 5

11.	Посчитать и вывести общее количество актёров, средний возраст, минимальный и максимальный возраст
SELECT 
count(*) as 'количество',
AVG(age) AS 'средний возраст',
MIN(age) AS 'минимальный возраст',
max(age) as 'максимальный возраст' 
FROM actors 


12.	Вывести страны, в которых более 5 и менее 10 актёров.
SELECT country,count(*) as 'количество актеров'
FROM actors 
GROUP by country
HAVING count(*) > 5 AND count(*) < 10
ORDER by count(*) DESC

13.	Посчитать средний возраст мужчин и женщин.
SELECT sex,AVG(age) AS 'средний возраст'
FROM actors 
WHERE sex is NOT NULL
group by sex

14.	Для каждой страны вывести самого молодого актёра.
SELECT country,name, last_name,min(age)
FROM actors 
WHERE age is NOT NULL
GROUP by country


15.	Вывести страны, в которых нет ни одной актрисы (female).

FROM actors
WHERE  country not in (
SELECT country
FROM actors
